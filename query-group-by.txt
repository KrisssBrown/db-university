SELECT COUNT(`enrolment_date`) AS `totale_iscritti_anno`
FROM `students`
GROUP BY YEAR(`enrolment_date`)
ORDER BY YEAR(`enrolment_date`) ASC;


SELECT COUNT(*), `office_address`
FROM `teachers`
GROUP BY `office_address`;


SELECT `exam_id`, SUM(`vote`) AS `somma_voti`, AVG(`vote`) AS `media_voti`
FROM `exam_student`
GROUP BY `exam_id`;


SELECT COUNT(`name`) AS `totale_corsi_dipartimento`, `department_id`
FROM `degrees`
GROUP BY `department_id`;
